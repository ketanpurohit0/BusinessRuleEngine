<pyodbc->PostgreSQL>
conn_str = (\
    "DRIVER={PostgreSQL ANSI};"\
    "DATABASE=postgres;"\
    "UID=postgres;"\
    "PWD=*secret*;"\
    "SERVER=localhost;"\
    "PORT=5432;"\
    )
conn = pyodbc.connect(conn_str)
crsr = conn.execute("SELECT * FROM FOO limit 50")
row = crsr.fetchone()
print(row)
crsr.close()
conn.close()
